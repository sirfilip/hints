# methods example
package main

import "fmt"

type Thing struct {
	Name string
}

func (t Thing) GetName() string {
	return t.Name
}

func (t *Thing) SetName(name string) {
	t.Name = name
}

func main() {
	t := Thing{"Filip"}
	fmt.Println(t.GetName())

	(&t).SetName("sirfilip")
	fmt.Println(t.GetName())
}
